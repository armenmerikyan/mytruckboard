try {
URL url = new URL("https://79vnubxva2.execute-api.us-east-1.amazonaws.com/prod/AddPreference");
HttpURLConnection httpCon = (HttpURLConnection) url.openConnection();
httpCon.setDoOutput(true);
httpCon.setRequestMethod("POST");
OutputStream os = httpCon.getOutputStream();
String randomCustomerID = UUID.randomUUID().toString();
String input = "{\"externalID\":f256dd9e-70c4-4717-89f0-feb55cc158ab, \"channelID\": \"ch88845\", \"brandID\": \"br34445\", \"productID\": \"pr4343445\", \"message\": \"CONTACT OK\"}";
os.write(input.toString().getBytes("UTF-8"));
os.flush();
os.close();
BufferedReader respBuf = new BufferedReader(
new InputStreamReader(httpCon.getInputStream(), "UTF-8"));
StringBuilder respData = new StringBuilder();
String jsonPartial = null;
while ((jsonPartial = respBuf.readLine()) != null) {
respData.append(jsonPartial + "\n");
}
respBuf.close();
System.out.print(randomCustomerID);
System.out.print(respData.toString());
} catch (Exception e) {
e.printStackTrace();
}
